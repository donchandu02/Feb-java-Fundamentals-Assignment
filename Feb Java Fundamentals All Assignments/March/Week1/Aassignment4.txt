package MarchWweek1;

public class Assi4 {

	public static void main(String[] args) {
		// binary to decimal.
		int n = 1011;
		int t =  n;
		int power = 0;
		int r = 0;
		while (t>0) {
			int d = t%10;
			t=t/10;
			 r = r + d*(int) Math.pow(2, power++);
		}
		System.out.println(r);

	}

}
