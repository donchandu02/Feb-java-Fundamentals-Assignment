package FebWeek2;

public class Assi5 {

	public static void main(String[] args) {
		// first 7 terms of N*6N which r divisible by 6.
		int count = 7;
		int temp = 0;
		
		for(int i=1;i<=count;i++) {
			temp = i*(6*i);
			if(temp%6==0) {
				System.out.print(temp+" ");
			}
		}

	}

}
